// JOIN with TABLE AND COLUMN ALIASES
// Demonstrates how the clause-specific placeholder architecture renders the same column object
// differently depending on the SQL clause:
// - SELECT clause: Renders with alias (e.g., [o].[OrderDate] AS [OrderedDate])
// - ORDER BY clause: Uses alias if set (e.g., [OrderedDate])
// - ON clause: Always shows full qualified reference, ignoring aliases (e.g., [p].[ItemNumber])
// This enables storing aliased column references and safely reusing them across clauses.
var p_join = Sql.GetTable<Product>();
var o = Sql.GetTable<Order>();
var orderQuantity = 10;

// Alias columns in SELECT; they'll render correctly in ORDER BY and other clauses
var aliasedJoinQuery = Sql.Build($@"
SELECT
  {p_join[prod => prod.ItemNumber].As("SKU")},
  {p_join[prod => prod.Name]},
  {o[ord => ord.OrderId].As("OrderNumber")},
  {o[ord => ord.OrderDate].As("OrderedDate")}
FROM {p_join.As("p")}
INNER JOIN {o.As("o")}
  ON {p_join[prod => prod.ItemNumber]} = {o[ord => ord.ProductItemNumber]}
WHERE {o[ord => ord.Quantity]} > {orderQuantity}
ORDER BY {o[ord => ord.OrderDate]} DESC
");
Console.WriteLine("JOIN with Table and Column Aliases (stored references - note: ON uses fresh refs):");
Console.WriteLine(aliasedJoinQuery);
Console.WriteLine();

/*
Output:
SELECT
  [p].[ItemNumber] AS [SKU],
  [p].[Name],
  [o].[OrderId] AS [OrderNumber],
  [o].[OrderDate] AS [OrderedDate]
FROM [dbo].[Product] AS [p]
INNER JOIN [dbo].[Orders] AS [o]
  ON [p].[ItemNumber] = [o].[ProductItemNumber]
WHERE [o].[Quantity] > @p0
ORDER BY [OrderedDate] DESC
*/

// Alternative: Fluent .On() method for explicit ON clause control
// Recommended when you want very clear separation between SELECT aliases and ON conditions.
// The On() method automatically passes fresh column references to the ON clause.
var fluentJoinQuery = Sql.Build($@"
SELECT 
  {p_join[prod => prod.ItemNumber].As("SKU")},
  {p_join[prod => prod.Name]},
  {o[ord => ord.OrderId].As("OrderNumber")},
  {o[ord => ord.OrderDate].As("OrderedDate")}
FROM {p_join.As("p").InnerJoin(o.As("o")).On(p_join[prod => prod.ItemNumber], o[ord => ord.ProductItemNumber])}
WHERE {o[ord => ord.Quantity]} > {orderQuantity}
ORDER BY {o[ord => ord.OrderDate]} DESC

Console.WriteLine("JOIN using fluent .On() method (clean, independent column references):");

// JOIN with AND/OR conditions - showing complex ON clause logic
// Mix AND and OR for flexible JOIN conditions
var highValue = 500m;

// Sql.Format() now automatically detects clause context from surrounding SQL
// No need to explicitly pass "ORDER BY" - it's detected from the surrounding query
var joinWithOrQuery = Sql.Build($@"

  {p_join[prod => prod.ItemNumber].As("SKU")},
  {p_join[prod => prod.Name]},
  {o[ord => ord.OrderDate].As("OrderedDate")}
FROM {p_join.As("p")
    .InnerJoin(o.As("o"))
    .On(p_join[prod => prod.ItemNumber], o[ord => ord.ProductItemNumber])
    .And(Sql.Format("{0} > {1}", o[ord => ord.OrderDate], minOrderDate))
    .Or(Sql.Format("{0} > {1}", p_join[prod => prod.Price], highValue))}
WHERE {o[ord => ord.Quantity]} > {orderQuantity}
ORDER BY {Sql.Format("{0} DESC", o[ord => ord.OrderDate])}
");

/// <summary>
/// Product DTO with attribute-based database mapping
/// Eliminates redundancy between DTO and schema definitions
/// </summary>
// [SqlTable("ReleasedProducts", "dbo")]
[SqlTable(schemaName: "dbo")]
public class Product
{
    /// <summary>Product item number (SKU)</summary>
    // [SqlColumn]
    public int ItemNumber { get; set; }

    /// <summary>Product name - mapped from database column 'ProductName'</summary>
    // [SqlColumn("ProductName")]
    public string Name { get; set; } = null!;

    /// <summary>Product price</summary>
    // [SqlColumn]
    public decimal Price { get; set; }
}

/// <summary>
/// Order DTO with attribute-based database mapping
/// Demonstrates JOIN usage with multiple DTOs
/// </summary>
[SqlTable("Orders", "dbo")]
public class Order
{
    /// <summary>Order ID (PK)</summary>
    [SqlColumn]
    public int OrderId { get; set; }

    /// <summary>Product item number (FK)</summary>
    [SqlColumn]
    public int ProductItemNumber { get; set; }

    /// <summary>Order date</summary>
    [SqlColumn]
    public DateTime OrderDate { get; set; }

    /// <summary>Order quantity</summary>
    [SqlColumn]
    public int Quantity { get; set; }
}

namespace SqlSchema
{
  public class ProductsTable(string? schema = null) : SqlTable("ReleasedProducts", schema)
  {
      private SqlColumn? _itemNumber;
      private SqlColumn? _productName;
      private SqlColumn? _price;

      public SqlColumn ItemNumber => _itemNumber ??= new(this, "ItemNumber");

      public SqlColumn ProductName => _productName ??= new(this, "ProductName");

      public SqlColumn Price => _price ??= new(this, "Price");
  }

  public static class Dbo
  {
      public const string Schema = "dbo";
      public static ProductsTable Products => new(Schema);
  }
